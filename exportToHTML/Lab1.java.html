19           }

29       /** Возвращает площадь треугольника, образованного тремя точками */
30       public static double computeArea(Point3d p1, Point3d p2, Point3d p3) {
31           double side_a = p1.distanceTo(p2);
32           double side_b = p1.distanceTo(p3);
33           double side_c = p2.distanceTo(p3);
34           double semi_p = (side_a + side_b + side_c) / 2;
35           return Math.sqrt(semi_p * (semi_p - side_a) * (semi_p - side_b) * (semi_p - side_c));
36       }
37   }
